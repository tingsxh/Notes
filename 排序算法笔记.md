package com.leetcode;

import java.util.HashMap;
import java.util.Map;

/*** 排序算法总结
 * @author d00464537
 * @created 2020/3/31
 */
public class LeetCodeSort {

    public static void main(String[] args) {
        LeetCodeSort leetCodeSort = new LeetCodeSort();
        int[] nums = {5,2,3,1};
        leetCodeSort.insertSort2(nums);
        System.out.println("--");
    }

    /***
     * 插入排序
     * 思路是双重循环，每个数都往前遍历一直到找到合适的位置
     * @param nums
     */
    private void insertSort(int[] nums) {
        for (int i = 0; i < nums.length - 1; i++) {
            for (int j = i + 1; j > 0; j--) {
                if (nums[j] < nums[j - 1]) {
                    // 交换
                    int temp = nums[j];
                    nums[j] = nums[j - 1];
                    nums[j - 1] = temp;
                } else {
                    break;
                }
            }
        }
    }

    /**
     * 插入排序改进版本
     *
     * @param nums
     */
    private void insertSort2(int[] nums) {
        for (int i = 0; i < nums.length - 1; i++) {
            int j = i + 1;
            // 记录需要插入的数字
            int temp = nums[j];
            for (; j > 0; j--) {
                if (temp < nums[j - 1]) {
                    // 数字依次往前挪
                    nums[j] = nums[j - 1];
                } else {
                    break;
                }
            }
            // 只交换一次
            nums[j] = temp;
        }
    }

    /***希尔排序
     *  思路其实和插入排序十分类似，只不过分成了多组数据同时进行插入排序
     * @param nums
     */
    private void shellSort(int[] nums) {
        // 这个变量代表着每一组两个数之间的距离
        int gap = nums.length / 2;
        while (gap > 0) {
            // 再对每一组进行划分
            for (int i = gap; i < nums.length; i++) {
                // 按照插入排序的思路，gap 就是第一组的第二个数，需要往前寻找到它合适的位置
                for (int j = i; j - gap >= 0; ) {
                    if (nums[j] < nums[j - gap]) {
                        // 如果比前一个数字小的h话，则进行交换
                        int temp = nums[j];
                        nums[j] = nums[j - gap];
                        nums[j - gap] = temp;
                        j -= gap;
                    } else {
                        break;
                    }
                }
            }
            // 一轮结束之后，继续缩小gap的值
            gap /= 2;
        }
    }

    /****
     * 归并排序
     * @param nums
     */
    private void mergeSort(int[] nums) {
        // 归并排序，这里最好用闭区间处理
        sort(0, nums.length - 1, nums);
    }

    private void sort(int start, int end, int[] nums) {
        if (start >= end) {
            return;
        }
        int mid = (start + end) / 2;
        sort(start, mid, nums);
        sort(mid + 1, end, nums);
        merge(start, mid, end, nums);
    }

    private void merge(int start, int mid, int end, int[] nums) {
        int[] left = new int[mid - start + 1];
        int[] right = new int[end - mid];

        // 左数组
        for (int i = start; i <= mid; i++) {
            left[i - start] = nums[i];
        }
        // 右数组
        for (int i = mid + 1; i <= end; i++) {
            right[i - (mid + 1)] = nums[i];
        }

        int l1 = 0, r1 = 0;
        // 合并
        for (int i = start; i <= end; i++) {
            if (l1 >= left.length) {
                nums[i] = right[r1++];
            } else if (r1 >= right.length) {
                nums[i] = left[l1++];
            } else if (left[l1] < right[r1]) {
                nums[i] = left[l1];
                l1++;
            } else {
                nums[i] = right[r1];
                r1++;
            }
        }
    }

    /****
     * 堆排序
     * @param nums
     */
    private void heapSort(int[] nums) {
        //堆排序从数组的n/2开始的
        for (int i = (nums.length - 1) / 2; i >= 0; i--) {
            heapShift(nums, i, nums.length);
        }
        // 将堆顶元素与最后一个交换，然后重新堆化数组
        for (int i = nums.length - 1; i > 0; i--) {
            int max = nums[0];
            nums[0] = nums[i];
            nums[i] = max;
            heapShift(nums, 0, i);
        }

    }

    /****
     * 数组堆化
     * @param nums
     * @param i
     * @param length
     */
    private void heapShift(int[] nums, int i, int length) {
        int left = 2 * i + 1;
        while (left < length) {
            if (left + 1 < length && nums[left + 1] > nums[left]) {
                left++;
            }
            // 如果
            if (nums[i] < nums[left]) {
                int temp = nums[left];
                nums[left] = nums[i];
                nums[i] = temp;
                i = left;
                left = 2 * i + 1;
            } else {
                break;
            }
        }
    }

    /***快速排序
     *
     * @param nums
     */
    private void quickSort(int[] nums) {
        sort2(0, nums.length - 1, nums);
    }

    private void sort2(int start, int end, int[] nums) {

        if (start >= end) {
            return;
        }
        int pos = getPos(start, end, nums);
        sort2(start, pos - 1, nums);
        sort2(pos + 1, end, nums);
    }

    /***获取中间位置信息
     *
     * @param start
     * @param end
     * @param nums
     * @return
     */
    private int getPos(int start, int end, int[] nums) {
        int v = nums[start];
        int pos = start;
        for (int i = start + 1; i <= end; i++) {
            if (nums[i] < v) {
                pos++;
                int temp = nums[i];
                nums[i] = nums[pos];
                nums[pos] = temp;
            }
        }
        nums[start] = nums[pos];
        nums[pos] = v;
        return pos;
    }

    // 计数排序
    private void countSort(int[] nums) {
        int max = Integer.MIN_VALUE;
        int min = Integer.MAX_VALUE;
        Map<Integer, Integer> map = new HashMap<>();
        for (int num : nums) {
            // 先收集每个数字出现的次数
            map.compute(num, (k, v) -> v == null ? 1 : v + 1);
            max = Math.max(max, num);
            min = Math.min(min, num);
        }

        // 再依次从小到大填充
        int index = 0;
        for (int i = min; i <= max; i++) {
            int size = map.getOrDefault(i, 0);
            while (size > 0) {
                nums[index++] = i;
                size--;
            }
        }

    }

}
